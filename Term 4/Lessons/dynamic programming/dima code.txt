import datetime

lst = [-1 for i in range(0, 100)]

def fib(n):

    if n <= 1:
        lst[n] = n
        return n
    else:
        if lst[n] != -1:
            num = lst[n]
        else:
            num = fib(n-1) + fib(n-2)
            lst[n] = num
        return num

start = datetime.datetime.now()

print('Число: ', fib(37))

end = datetime.datetime.now()

print(str(end - start).split(':')[2])

print(lst)
